# AI Patents Chart — Assignment A02 Starter
# Replace sample_data with real data from USPTO or WIPO

import matplotlib.pyplot as plt

# Sample data — replace with real patent counts per year
years  = [2014, 2015, 2016, 2017, 2018, 2019, 2020, 2021, 2022, 2023]
patents = [3100, 4200, 5800, 8400, 12000, 16500, 22000, 30000, 41000, 55000]

fig, ax = plt.subplots(figsize=(10, 6))
ax.bar(years, patents, color='#4299e1', edgecolor='#2b6cb0', linewidth=0.8)

# Labels and formatting
ax.set_title('AI Patents Issued Per Year (2014–2023)', fontsize=14, fontweight='bold')
ax.set_xlabel('Year', fontsize=11)
ax.set_ylabel('Number of Patents Filed', fontsize=11)
ax.set_xticks(years)
ax.tick_params(axis='x', rotation=45)

# Source annotation
ax.annotate('Source: USPTO / WIPO AI Patent Data',
            xy=(0.01, 0.01), xycoords='axes fraction',
            fontsize=8, color='gray')

plt.tight_layout()
plt.savefig('ai_patents_chart.png', dpi=150)  # save for insertion into Word/PPT
plt.show()
