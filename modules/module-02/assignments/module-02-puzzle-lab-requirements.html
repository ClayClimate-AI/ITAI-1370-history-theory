# Google puzzles often test search/optimization logic —
# the same logic AI uses for pathfinding and decision-making.
# This shows how a brute-force vs smart search differs.

from itertools import permutations

def brute_force_search(prisoners, boxes):
    """Try every possible path — O(n!) — what AI avoids."""
    for perm in permutations(range(len(boxes))):
        if check_strategy(prisoners, boxes, perm):
            return perm
    return None

def cycle_following_strategy(prisoner_num, boxes, limit):
    """
    The optimal strategy: follow cycles.
    Each prisoner starts at their own number box,
    then follows the number inside — like a linked list.
    AI uses this kind of heuristic to avoid brute force.
    """
    current = prisoner_num
    for _ in range(limit):
        current = boxes[current]   # follow the chain
        if current == prisoner_num:
            return True           # found own number in time
    return False                   # ran out of attempts

# Key insight for Task 2:
# AI search problems use heuristics (smart shortcuts)
# rather than brute force — this puzzle demonstrates why.
